Результат →

Створити функцію знаходження мінімального і максимального елемента масиву getMinAndMax(array). Результат повинен виводитись у консоль.

Вихідний масив:

Результат →


    
Код:

    
let srcArr2 = [432, 12, 634, 2367, 12, 2, 452, 95923];
document.getElementById('srcArr2').innerHTML = `[${srcArr2.join(', ')}]`
function getMinAndMax() {
    let min = Infinity;
    let max = -Infinity;
    for (let i = 0; i <= srcArr2.length; i++) {
        if (srcArr2[i] < min) {
            min = srcArr2[i];
        } else if (srcArr2[i] >= max) {
            max = srcArr2[i];
        }
    }
    console.log(`min = ${min}, max = ${max}`);
    document.getElementById('result2').append(`> min = ${min}, max = ${max}\n`);
}
